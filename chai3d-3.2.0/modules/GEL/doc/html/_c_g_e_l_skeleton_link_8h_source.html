116  inline void computeForces()
117  {
118  // update basic parameters of current link
119  m_wLink01 = cSub(m_node1->m_pos, m_node0->m_pos);
120  m_wLink10 = cMul(-1, m_wLink01);
121  m_length = m_wLink01.length();
122 
123  m_wzLink01 = cMul((m_node0->m_rot), m_nzLink01);
124  m_wzLink10 = cMul((m_node1->m_rot), m_nzLink10);
125 
126  //-------------------------------------------------------------
127  // ELONGATION:
128  //-------------------------------------------------------------
129  // if distance too small, no forces are applied
130  if (m_length < 0.000001) { return; }
131 
132  // elongation compute force
133  double f = m_kSpringElongation * (m_length - m_length0);
134 
135  // apply force
136  if (m_length > 0.000001)
137  {
138  chai3d::cVector3d force = cMul(f/m_length, m_wLink01);
139  m_node0->addForce(force);
140  chai3d::cVector3d tmpfrc = cMul(-1, force);
141  m_node1->addForce(tmpfrc);
142  }
143 
144  //-------------------------------------------------------------
145  // FLEXION:
146  //-------------------------------------------------------------
147  // compute flexion forces and torques
148  chai3d::cVector3d torqueDir0 = cCross(m_wzLink01, m_wLink01);
149  chai3d::cVector3d torqueDir1 = cCross(m_wzLink10, m_wLink10);
150  double angle0 = cAngle(m_wzLink01, m_wLink01);
151  double angle1 = cAngle(m_wzLink10, m_wLink10);
152  double torqueMag0 = angle0 * m_kSpringFlexion;
153  double torqueMag1 = angle1 * m_kSpringFlexion;
154 
155  if ((m_length > 0.000001) && (m_enabled))
156  {
157  if (torqueMag0 > 0.0001)
158  {
159  chai3d::cVector3d torque0 = cMul(torqueMag0, cNormalize(torqueDir0));
160  m_node0->addTorque(torque0);
161  chai3d::cVector3d force1 = cMul((torqueMag0/m_length), cNormalize(cCross(m_wLink01, torque0)));
162  m_node1->addForce(force1);
163  chai3d::cVector3d tmpfrc = cMul(-1,force1);
164  m_node0->addForce(tmpfrc);
165  }
166 
167  if (torqueMag1 > 0.0001)
168  {
169  chai3d::cVector3d torque1 = cMul(torqueMag1, cNormalize(torqueDir1));
170  m_node1->addTorque(torque1);
171  chai3d::cVector3d force0 = cMul((torqueMag1/m_length), cNormalize(cCross(m_wLink10, torque1)));
172  m_node0->addForce(force0);
173  chai3d::cVector3d tmpfrc = cMul(-1,force0);
174  m_node1->addForce(tmpfrc);
175  }
176  }
177 
178  //-----------------------------------------------------------------------
179  // TORSION:
180  //-----------------------------------------------------------------------
181 
182  if (m_enabled)
183  {
184  // update frame vectors in world coordinates.
185  m_node0->m_rot.mulr(m_A0, m_wA0);
186  m_node0->m_rot.mulr(m_B0, m_wB0);
187  m_node1->m_rot.mulr(m_A1, m_wA1);
188 
189  // compute torsional angle and torque
190  chai3d::cVector3d v0, v1;
191  m_wA0.crossr(m_wLink01, v0);
192  m_wA1.crossr(m_wLink01, v1);
193  v0.normalize();
194  v1.normalize();
195 
196  chai3d::cVector3d torque;
197  v0.crossr(v1, m_torsionAxisAngle);
198  m_torsionAxisAngle.mulr(m_kSpringTorsion, torque);
199 
200  m_node0->addTorque(torque);
201  chai3d::cVector3d tmptrq = -1.0 * torque;
202  m_node1->addTorque(tmptrq);
203  }
204  }
