118  inline void computeNextPose(double a_timeInterval)
119  {
120  if (!m_fixed)
121  {
122  // Euler double integration for position
123  chai3d::cVector3d damping;
124  m_vel.mulr(-m_kDampingPos * m_mass, damping);
125  m_force.add(damping);
126  m_acc = cDiv(m_mass, cAdd(m_force, m_externalForce));
127  m_vel = cAdd(m_vel, cMul(a_timeInterval, m_acc));
128  m_nextPos = cAdd(m_pos, cMul(a_timeInterval, m_vel));
129  }
130  else
131  {
132  m_nextPos = m_pos;
133  }
134 
135  // Euler double integration for rotation
136  chai3d::cVector3d dampingAng;
137  m_angVel.mulr(-m_kDampingRot * m_mass, dampingAng);
138  m_torque.add(dampingAng);
139  m_angAcc = cMul((1/m_inertia), m_torque);
140  m_angVel = cAdd(m_angVel, cMul(a_timeInterval, m_angAcc));
141 
142  double normAngVel = m_angVel.length();
143  if (normAngVel < 0.000001)
144  {
145  m_nextRot = m_rot;
146  }
147  else
148  {
149  m_nextRot = m_rot;
150  m_nextRot.rotateAboutGlobalAxisRad(m_angVel, a_timeInterval * normAngVel);
151  }
152  }
153 
155  inline void applyNextPose()
156  {
157  m_pos = m_nextPos;
158  m_rot = m_nextRot;
159  }
160 
162  inline void clearForces()
163  {

168  }
169  else
170  {
171  m_force.zero();
172  }
173  m_torque.zero();
174  }
175 
177  inline void clearExternalForces()

180  m_externalTorque.zero();
181  }

184  inline void render()
185  {
186  // set pose
187  chai3d::cTransform mat;
188  mat.set(m_pos, m_rot);
189  glPushMatrix();
190  glMultMatrixd( (const double *)mat.getData() );
191 
192  // draw node
193  m_color.render();
194  chai3d::cDrawSphere(m_radius, 12, 12);
195 
196  // draw frame
197  if (s_default_showFrame == true)
198  {
199  double frameScale = 3.0 * m_radius;
200  chai3d::cDrawFrame(frameScale);
201  }
202 
203  // pop OpenGL matrix
204  glPopMatrix();
205 
206  // render external forces
207  glColor4fv( (const float *)&m_color);
208  chai3d::cVector3d v = cAdd(m_pos, cMul(1.0/50.0, m_externalForce));
209  glBegin(GL_LINES);
210  glVertex3dv( (const double *)&m_pos);
211  glVertex3dv( (const double *)&v);
212  glEnd();
213  }
