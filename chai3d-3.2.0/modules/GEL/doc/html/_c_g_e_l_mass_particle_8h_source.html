106  inline void computeNextPose(double a_timeInterval)
107  {
108  if (!m_fixed)
109  {
110  // Euler double integration for position
111  chai3d::cVector3d damping;
112  m_vel.mulr(-m_kDampingPos * m_mass, damping);
113  m_force.add(damping);
114  m_acc = cDiv(m_mass, cAdd(m_force, m_externalForce));
115  m_vel = cAdd(m_vel, cMul(a_timeInterval, m_acc));
116  m_nextPos = cAdd(m_pos, cMul(a_timeInterval, m_vel));
117  }
118  else
119  {
120  m_nextPos = m_pos;
121  }
122  }
123 
125  inline void applyNextPose()
126  {
127  m_pos = m_nextPos;
128  }
129 
131  inline void clearForces()
132  {
133  if (m_useGravity)
134  {
135  m_gravity.mulr(m_mass, m_force);
136  }
137  else
138  {
139  m_force.zero();
140  }
141  }
142 
144  inline void clearExternalForces()
145  {
146  m_externalForce.zero();
147  }
148 
150  inline void render()
151  {
152  // draw point
153  m_color.render();
154  glBegin(GL_POINTS);
155  glVertex3dv( (const double *)&m_pos);
156  glEnd();
157 
158  // render external forces
159  glColor4fv( (const float *)&m_color);
160  chai3d::cVector3d v = cAdd(m_pos, cMul(s_scale_force_vector_display, m_externalForce));
161  glBegin(GL_LINES);
162  glVertex3dv( (const double *)&m_pos);
163  glVertex3dv( (const double *)&v);
164  glEnd();
165  }
