115  inline void computeForces()
116  {
117  // update basic parameters of current link
118  chai3d::cVector3d m_link01 = cSub(m_node1->m_pos, m_node0->m_pos);
119  double m_length = m_link01.length();
120 
121  //-------------------------------------------------------------
122  // ELONGATION:
123  //-------------------------------------------------------------
124  // if distance too small, no forces are applied
125  if (m_length < 0.000001) { return; }
126 
127  // elongation compute force
128  double f = m_kSpringElongation * (m_length - m_length0);
129 
130  // apply force
131  if (m_length > 0.000001)
132  {
133  chai3d::cVector3d force = cMul(f/m_length, m_link01);
134  m_node0->addForce(force);
135  chai3d::cVector3d tmpfrc = cMul(-1, force);
136  m_node1->addForce(tmpfrc);
137  }
138  }
